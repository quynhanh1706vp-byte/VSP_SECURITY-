GitHub Actions — vsp-scan.yml
Copy vào .github/workflows/ trong repo của bạn
# .github/workflows/vsp-scan.yml name: VSP Security Scan on: push: branches: [ main, develop ] pull_request: branches: [ main ] jobs: vsp-scan: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - name: Trigger VSP scan id: vsp run: | RESULT=$(curl -sf -X POST ${{ secrets.VSP_URL }}/api/v1/vsp/run \ -H "Authorization: Bearer ${{ secrets.VSP_TOKEN }}" \ -H "Content-Type: application/json" \ -d '{"mode":"FULL","profile":"FAST","src":"."}') RID=$(echo $RESULT | jq -r '.rid') echo "rid=$RID" >> $GITHUB_OUTPUT # Poll until done for i in $(seq 1 30); do STATUS=$(curl -sf \ -H "Authorization: Bearer ${{ secrets.VSP_TOKEN }}" \ ${{ secrets.VSP_URL }}/api/v1/vsp/run/$RID | jq -r '.status') [ "$STATUS" = "DONE" ] && break sleep 10 done - name: Check gate run: | GATE=$(curl -sf \ -H "Authorization: Bearer ${{ secrets.VSP_TOKEN }}" \ ${{ secrets.VSP_URL }}/api/v1/vsp/run/${{ steps.vsp.outputs.rid }} \ | jq -r '.gate') echo "Gate: $GATE" [ "$GATE" = "FAIL" ] && exit 1 || exit 0 - name: Upload SARIF if: always() uses: github/codeql-action/upload-sarif@v3 with: sarif_file: vsp-results.sarif

Jenkins — Jenkinsfile
Add to repo root · works with declarative + scripted pipeline
// Jenkinsfile — declarative pipeline with VSP gate pipeline { agent any environment { VSP_URL = credentials('vsp-url') VSP_TOKEN = credentials('vsp-token') } stages { stage('VSP Security Scan') { steps { sh ''' set -e RID=$(curl -sf -X POST $VSP_URL/api/v1/vsp/run \ -H "Authorization: Bearer $VSP_TOKEN" \ -H "Content-Type: application/json" \ -d '{"mode":"FULL","profile":"FAST","src":"."}' | jq -r '.rid') echo "VSP run: $RID" for i in $(seq 1 30); do S=$(curl -sf -H "Authorization: Bearer $VSP_TOKEN" \ $VSP_URL/api/v1/vsp/run/$RID | jq -r '.status') [ "$S" = "DONE" ] && break sleep 10 done GATE=$(curl -sf -H "Authorization: Bearer $VSP_TOKEN" \ $VSP_URL/api/v1/vsp/run/$RID | jq -r '.gate') echo "Gate: $GATE" [ "$GATE" = "FAIL" ] && exit 1 || exit 0 ''' } } } post { always { archiveArtifacts artifacts: 'vsp-results.sarif', allowEmptyArchive: true } } }

vspcli — Local / shell pre-commit hook
For developer workstations or any custom CI runner
#!/usr/bin/env bash # vspcli wrapper — works in any CI runner or local dev machine # Save as: scripts/vsp-scan.sh (chmod +x) set -euo pipefail # Config (override via env) VSP_URL="${VSP_URL:-http://127.0.0.1:8921}" VSP_TOKEN="${VSP_TOKEN:?required}" SCAN_MODE="${SCAN_MODE:-FULL}" SCAN_PROFILE="${SCAN_PROFILE:-FAST}" SRC="${SRC:-.}" # Trigger RID=$(curl -sf -X POST "$VSP_URL/api/v1/vsp/run" \ -H "Authorization: Bearer $VSP_TOKEN" \ -H "Content-Type: application/json" \ -d "{\"mode\":\"$SCAN_MODE\",\"profile\":\"$SCAN_PROFILE\",\"src\":\"$SRC\"}" \ | jq -r '.rid') echo "VSP run: $RID" # Wait for i in $(seq 1 60); do S=$(curl -sf -H "Authorization: Bearer $VSP_TOKEN" \ "$VSP_URL/api/v1/vsp/run/$RID" | jq -r '.status') [ "$S" = "DONE" ] && break sleep 5 done # Gate verdict GATE=$(curl -sf -H "Authorization: Bearer $VSP_TOKEN" \ "$VSP_URL/api/v1/vsp/run/$RID" | jq -r '.gate') echo "Gate: $GATE" [ "$GATE" = "FAIL" ] && exit 1 || exit 0
