GitHub Actions — vsp-scan.yml
Copy vào .github/workflows/ trong repo của bạn
# .github/workflows/vsp-scan.yml name: VSP Security Scan on: push: branches: [ main, develop ] pull_request: branches: [ main ] jobs: vsp-scan: runs-on: ubuntu-latest steps: - uses: actions/checkout@v4 - name: Trigger VSP scan id: vsp run: | RESULT=$(curl -sf -X POST ${{ secrets.VSP_URL }}/api/v1/vsp/run \ -H "Authorization: Bearer ${{ secrets.VSP_TOKEN }}" \ -H "Content-Type: application/json" \ -d '{"mode":"FULL","profile":"FAST","src":"."}') RID=$(echo $RESULT | jq -r '.rid') echo "rid=$RID" >> $GITHUB_OUTPUT # Poll until done for i in $(seq 1 30); do STATUS=$(curl -sf \ -H "Authorization: Bearer ${{ secrets.VSP_TOKEN }}" \ ${{ secrets.VSP_URL }}/api/v1/vsp/run/$RID | jq -r '.status') [ "$STATUS" = "DONE" ] && break sleep 10 done - name: Check gate run: | GATE=$(curl -sf \ -H "Authorization: Bearer ${{ secrets.VSP_TOKEN }}" \ ${{ secrets.VSP_URL }}/api/v1/vsp/run/${{ steps.vsp.outputs.rid }} \ | jq -r '.gate') echo "Gate: $GATE" [ "$GATE" = "FAIL" ] && exit 1 || exit 0 - name: Upload SARIF if: always() uses: github/codeql-action/upload-sarif@v3 with: sarif_file: vsp-results.sarif
